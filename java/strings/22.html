Program to find maximum and minimum occurring character in a string

Code:-



            import java.util.HashMap;
            import java.util.Map;
            class a {
            
                public static void main(String[] args) {
                    String str = "GeeksforGeeks";
                    findMaxAndMinCountCharacters(str);
                }
            
                public static void findMaxAndMinCountCharacters(String str) {
                    HashMap countMap = new HashMap<>();
                    char[] characters = str.toCharArray();
            
                    for (char c : characters) {
                        if (c != ' ') {
                            countMap.put(c, countMap.getOrDefault(c, 0) + 1);
                        }
                    }
            
                    char maxOccurringChar = ' ';
                    char minOccurringChar = ' ';
                    int maxCount = -1;
                    int minCount = 1000000000;
            
                    for (Map.Entry entry : countMap.entrySet()) {
                        if (entry.getValue() > maxCount) {
                            maxCount = entry.getValue();
                            maxOccurringChar = entry.getKey();
                        }
            
                        if (entry.getValue() < minCount) {
                            minCount = entry.getValue();
                            minOccurringChar = entry.getKey();
                        }
                    }
            
                    System.out.println("Character with maximum occurrences: " + maxOccurringChar);
                    System.out.println("Character with minimum occurrences: " + minOccurringChar);
                }
            }
