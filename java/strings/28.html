Program to find the longest repeating sequence in a string

Code:-



            import java.util.HashMap;
            import java.util.Map;
            
            public class Main {
            
                public static void main(String[] args) {
                    String str = "abcabcabcabc";
                    System.out.println("Length of longest repeating sequence is " + longestRepeatingSequence(str));
                }
            
                public static int longestRepeatingSequence(String str) {
                    int len = str.length();
                    int maxCount = 0;
                    Map countMap = new HashMap<>();
            
                    for (int i = 0; i < len; i++) {
                        String seq = "";
                        for (int j = i; j < len; j++) {
                            seq += str.charAt(j);
                            int count = countMap.getOrDefault(seq, 0) + 1;
                            countMap.put(seq, count);
            
                            if (count > 1) {
                                maxCount = Math.max(maxCount, seq.length());
                            }
                        }
                    }
            
                    return maxCount;
                }
            }
