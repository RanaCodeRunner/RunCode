How to convert Object to String

Code:-



            class a {
                public static void main(String[] args) {
                    // Example: Convert Object to String
                    MyObject myObject = new MyObject("Hello, Object!");
            
                    // Method 1: Using toString() method
                    String str1 = myObject.toString();
                    System.out.println("Method 1: Using toString() - Result: " + str1);
            
                    // Method 2: Implicitly calling toString() (not explicitly required)
                    String str2 = myObject + "";
                    System.out.println("Method 2: Implicitly calling toString() - Result: " + str2);
                }
            }
            
            class MyObject {
                private String message;
            
                public MyObject(String message) {
                    this.message = message;
                }
            
                // Override the toString() method to provide a meaningful representation
                @Override
                public String toString() {
                    return "MyObject{" +
                            "message='" + message + '\'' +
                            '}';
                }
            }
