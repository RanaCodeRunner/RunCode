How to convert Timestamp to Date

Code:-



            import java.sql.Timestamp;
            import java.util.Date;
            
            public class TimestampToDateExample {
                public static void main(String[] args) {
                    // Example: Convert Timestamp to Date
                    Timestamp timestamp = new Timestamp(System.currentTimeMillis());
            
                    // Convert Timestamp to Date using the constructor
                    Date date1 = new Date(timestamp.getTime());
                    System.out.println("Converted Date (using constructor): " + date1);
            
                    // Convert Timestamp to Date using the toInstant() method
                    Date date2 = Date.from(timestamp.toInstant());
                    System.out.println("Converted Date (using toInstant()): " + date2);
                }
            }
